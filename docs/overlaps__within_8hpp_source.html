90 matches.clear();
91 if (subject.root_children == 0) {
92 return;
93 }
94
95 // This is the only place that we need to consider the min_overlap; if an
96 // subject interval envelops the query, it should be at least as wide, so
97 // will automatically satisfy min_overlap.
98 Position_ query_width = query_end - query_start;
99 if (params.min_overlap > 0 && query_width < params.min_overlap) {
100 return;
101 }
102
103 // If a subject interval doesn't satisfy these requirements, none of its
104 // children will either, so we can safely declare that iteration as being
105 // finished. Don't check the max_gap constraints here, as parent could fail
106 // this while its child could satisfy it.
107 auto is_finished = [&](Position_ subject_start) -> bool {
108 return subject_start > query_start;
109 };
110
111 // We start from the first subject interval that finishes at or after the query.
112 auto find_first_child = [&](Index_ children_start, Index_ children_end) -> Index_ {
113 auto ebegin = subject.ends.begin();
114 auto estart = ebegin + children_start;
115 auto eend = ebegin + children_end;
116 if (query_width == 0) {
117 // Unless the query is zero-width, in which case we're not interested in subjects that have the same endpoint.
118 return std::upper_bound(estart, eend, query_end) - ebegin;
119 } else {
120 return std::lower_bound(estart, eend, query_end) - ebegin;
121 }
122 };
123
124 Index_ root_child_at = find_first_child(0, subject.root_children);
125
126 workspace.history.clear();
127 while (1) {
128 Index_ current_subject;
129 if (workspace.history.empty()) {
130 if (root_child_at == subject.root_children || is_finished(subject.starts[root_child_at])) {
131 break;
132 }
133 current_subject = root_child_at;
134 ++root_child_at;
135 } else {
136 auto& current_state = workspace.history.back();
137 if (current_state.child_at == current_state.child_end || is_finished(subject.starts[current_state.child_at])) {
138 workspace.history.pop_back();
139 continue;
140 }
141 current_subject = current_state.child_at;
142 ++(current_state.child_at); // do this before the emplace_back(), otherwise the history might get reallocated and the reference would be dangling.
143 }
144
145 const auto& current_node = subject.nodes[current_subject];
146
147 // If max_gap is violated, we don't bother to add the current subject interval,
148 // but the children could be okay so we proceed to the next level of the NClist.
149 bool add_self = true;
150 if (params.max_gap.has_value()) {
151 auto subject_start = subject.starts[current_subject];
152 auto subject_end = subject.ends[current_subject];
153 auto subject_width = subject_end - subject_start;
154 if (subject_width - query_width > *(params.max_gap)) {
155 add_self = false;
156 }
157 }
158
159 if (add_self) {
160 matches.push_back(current_node.id);
161 if (params.quit_on_first) {
162 return;
163 }
164 if (current_node.duplicates_start != current_node.duplicates_end) {
165 matches.insert(matches.end(), subject.duplicates.begin() + current_node.duplicates_start, subject.duplicates.begin() + current_node.duplicates_end);
166 }
167 }
168
169 if (current_node.children_start != current_node.children_end) {
170 Index_ start_pos = find_first_child(current_node.children_start, current_node.children_end);
171 if (start_pos != current_node.children_end) {
172 workspace.history.emplace_back(start_pos, current_node.children_end);
173 }
174 }
175 }

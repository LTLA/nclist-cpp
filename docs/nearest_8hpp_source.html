60template<typename Index_, typename Position_>
61Index_ nearest_overlaps(
62 const Nclist<Index_, Position_>& subject,
63 const Position_ query_start,
64 const Position_ query_end,
65 const NearestParameters<Position_>& params,
66 NearestWorkspace<Index_>& workspace,
67 std::vector<Index_>& matches)
68{
69 const auto find_first_child = [&](const Index_ children_start, const Index_ children_end) -> Index_ {
70 const auto ebegin = subject.ends.begin();
71 const auto estart = ebegin + children_start;
72 const auto eend = ebegin + children_end;
73 return std::upper_bound(estart, eend, query_start) - ebegin;
74 };
75
76 const auto can_skip_search = [&](const Position_ subject_start) -> bool {
77 return subject_start > query_start;
78 };
79
80 const auto is_finished = [&](const Position_ subject_start) -> bool {
81 return subject_start >= query_end;
82 };
83
84 Index_ root_child_at = 0;
85 const bool root_skip_search = can_skip_search(subject.starts[0]);
86 if (!root_skip_search) {
87 root_child_at = find_first_child(0, subject.root_children);
88 }
89
90 workspace.history.clear();
91 while (1) {
92 Index_ current_subject;
93 bool skip_search;
94 if (workspace.history.empty()) {
95 if (root_child_at == subject.root_children || is_finished(subject.starts[root_child_at])) {
96 break;
97 }
98 current_subject = root_child_at;
99 skip_search = root_skip_search;
100 ++root_child_at;
101 } else {
102 auto& current_state = workspace.history.back();
103 if (current_state.child_at == current_state.child_end || is_finished(subject.starts[current_state.child_at])) {
104 workspace.history.pop_back();
105 continue;
106 }
107 current_subject = current_state.child_at;
108 skip_search = current_state.skip_search;
109 ++(current_state.child_at); // do this before the emplace_back(), otherwise the history might get reallocated and the reference would be dangling.
110 }
111
112 const auto& current_node = subject.nodes[current_subject];
113
114 matches.push_back(current_node.id);
115 if (params.quit_on_first) {
116 break;
117 }
118 if (current_node.duplicates_start != current_node.duplicates_end) {
119 matches.insert(matches.end(), subject.duplicates.begin() + current_node.duplicates_start, subject.duplicates.begin() + current_node.duplicates_end);
120 }
121
122 if (current_node.children_start != current_node.children_end) {
123 if (skip_search) {
124 workspace.history.emplace_back(current_node.children_start, current_node.children_end, true);
125 } else {
126 const Index_ start_pos = find_first_child(current_node.children_start, current_node.children_end);
127 if (start_pos != current_node.children_end) {
128 workspace.history.emplace_back(start_pos, current_node.children_end, can_skip_search(subject.starts[start_pos]));
129 }
130 }
131 }
132 }
133
134 return root_child_at;
135}

216template<typename Index_, typename Position_>
217void nearest(
218 const Nclist<Index_, Position_>& subject,
219 const Position_ query_start,
220 const Position_ query_end,
221 const NearestParameters<Position_>& params,
222 NearestWorkspace<Index_>& workspace,
223 std::vector<Index_>& matches)
224{
225 matches.clear();
226 if (subject.root_children == 0) {
227 return;
228 }
229
230 const auto root_index = nearest_overlaps(subject, query_start, query_end, params, workspace, matches);
231 if (!matches.empty()) {
232 return;
233 }
234
235 /****************************************
236 * If there are no overlaps, checking the distance to the first subject interval before or after the query.
237 *
238 * 'root_index' is initially defined in nearest_overlaps() as the upper bound of 'query_start' on the subject ends of the root node of the NCList.
239 * If there are no overlaps, this definition won't change and the same value will be returned to nearest().
240 * At this point, 'root_index' defines the first subject interval on the root node that starts after the query.
241 * Conversely, 'root_index - 1' is the last subject interval on the root node that ends before the query.
242 *
243 * Thus, to find the nearest following interval, only the 'root_index'-th interval on the root node needs to be considered as it will have the start closest to 'query_end'.
244 * All later intervals on the root node will have later starts, otherwise they would be nested within the 'root_index`-th interval.
245 * We still need to check the descendents of the 'root_index'-th interval but we only need to care about the first child at each level of the lineage.
246 * In particular, we check whether it has the same start coordinate as its parent;
247 * if not, we know that the rest of that lineage will start after the 'root_index'-th interval and we terminate the search.
248 *
249 * Similarly, to find the nearest preceding interval, only the 'root_index - 1'-th interval on the root node needs to be considered as it will have the end closest to 'query_start'.
250 * All earlier intervals on the root node will have earlier ends, otherwise they would be nested within the 'root_index - 1`-th interval.
251 * We still need to check the descendents of the 'root_index - 1'-th interval but we only need to care about the last child at each level of the lineage.
252 * In particular, we check whether it has the same end coordinate as its parent;
253 * if not, we know that the rest of that lineage will end before the 'root_index - 1'-th interval and we terminate the search.
254 *
255 * Note that this function behaves a little strangely when a zero-width query is supplied and there is an identical zero-width subject interval.
256 * Specifically, the latter may be treated as preceding or following the former, depending on whether they are nested in other subject intervals.
257 * Fortunately, this doesn't have any bearing on the final outcome as it will still the nearest interval to the query in either case.
258 *
259 ****************************************/
260
261 std::optional<Position_> to_previous, to_next;
262 if (root_index) {
263 to_previous = query_start - subject.ends[root_index - 1];
264 }
265 if (root_index < subject.root_children) {
266 to_next = subject.starts[root_index] - query_end;
267 }
268
269 if (!to_next.has_value() || (to_previous.has_value() && *to_previous <= *to_next)) {
270 nearest_before(subject, root_index - 1, params, matches);
271 if (!matches.empty() && params.quit_on_first) {
272 return;
273 }
274 }
275 if (!to_previous.has_value() || (to_next.has_value() && *to_next <= *to_previous)) {
276 nearest_after(subject, root_index, params, matches);
277 if (!matches.empty() && params.quit_on_first) {
278 return;
279 }
280 }
281}

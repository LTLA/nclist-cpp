74// We template by overlap mode for compile-time distinction between the
75// different choices, somewhat for efficiency but also to allow us to do
76// compile-time checks for invalid operations that span multiple modes.
77enum class OverlapsAnyMode : char { BASIC, MIN_OVERLAP, MAX_GAP };

79template<OverlapsAnyMode mode_, typename Index_, typename Position_>
80void overlaps_any_internal(
81 const Nclist<Index_, Position_>& subject,
82 Position_ query_start,
83 Position_ query_end,
84 const OverlapsAnyParameters<Position_>& params,
85 OverlapsAnyWorkspace<Index_>& workspace,
86 std::vector<Index_>& matches)
87{
88 matches.clear();
89 if (subject.root_children == 0) {
90 return;
91 }
92 if constexpr(mode_ == OverlapsAnyMode::MIN_OVERLAP) {
93 if (query_end - query_start < params.min_overlap) {
94 return;
95 }
96 }

98 // If an subject interval doesn't satisfy these requirements, none of its
99 // children will either, so we can safely declare that iteration is
100 // finished. This allows us to skip a section of the NCList.
101 auto is_finished = [&](Position_ subject_start) -> bool {
102 if constexpr(mode_ == OverlapsAnyMode::MAX_GAP) {
103 if (subject_start < query_end) {
104 return false;
105 }
106 return subject_start - query_end > *(params.max_gap); // not >=, as a gap-extended end is inclusive to the starts.
107 } else if constexpr(mode_ == OverlapsAnyMode::MIN_OVERLAP) {
108 if (subject_start >= query_end) {
109 return true;
110 }
111 return query_end - subject_start < params.min_overlap;
112 } else {
113 return subject_start >= query_end;
114 }
115 };
116
117 typename std::conditional<mode_ == OverlapsAnyMode::BASIC, const char*, Position_>::type effective_query_start; // make sure compiler complains if we use this in basic mode.
118 if constexpr(mode_ == OverlapsAnyMode::MAX_GAP) {
119 // When a max gap is specified, we push the query start to the left
120 // so as to capture subject intervals within the specified gap range.
121 effective_query_start = safe_subtract_gap(query_start, *(params.max_gap));
122 } else if constexpr(mode_ == OverlapsAnyMode::MIN_OVERLAP) {
123 // When a minimum overlap is specified, we push the query start to the right
124 // so as to exclude subject intervals that end before the overlap is satisfied.
125 constexpr Position_ maxed = std::numeric_limits<Position_>::max();
126 if (maxed - params.min_overlap < query_start) {
127 // No point continuing as nothing will be found in the binary search.
128 return;
129 } else {
130 effective_query_start = query_start + params.min_overlap;
131 }
132 }
133
134 // The logic is that, if a query start precedes the subject start,
135 // the binary search is unnecessary as we always have to start at the first child
136 // (as subject_end >= subject_start >(=) query_start). Not only that, but
137 // the binary search can also be skipped for all of the grandchildren's
138 // children, which must necessarily start at or after the child.
139 auto is_query_preceding = [&](Position_ subject_start) -> bool {
140 if constexpr(mode_ == OverlapsAnyMode::BASIC) {
141 return subject_start > query_start;
142 } else {
143 return subject_start >= effective_query_start;
144 }
145 };
146
147 auto find_first_child = [&](Index_ children_start, Index_ children_end) -> Index_ {
148 auto ebegin = subject.ends.begin();
149 auto estart = ebegin + children_start;
150 auto eend = ebegin + children_end;
151 if constexpr(mode_ == OverlapsAnyMode::BASIC) {
152 // We use an upper bound as the interval ends are assumed to be non-inclusive,
153 // so we want the first subject end that is greater than the query_start.
154 return std::upper_bound(estart, eend, query_start) - ebegin;
155 } else {
156 // For gapped searches, we use a lower bound as the gap-subtracted
157 // start is inclusive with the interval ends. Consider a gap of
158 // zero; a query that is exactly contiguous with a subject interval
159 // will be considered as overlapping that interval.
160 //
161 // For minimum overlap searches, the overlap-added start represents
162 // the earliest subject end that satisfies the overlap. Now that
163 // we're comparing ends to ends, we can use a lower bound.
164 return std::lower_bound(estart, eend, effective_query_start) - ebegin;
165 }
166 };
167
168 Index_ root_child_at = 0;
169 bool root_skip_search = is_query_preceding(subject.starts[0]);
170 if (!root_skip_search) {
171 root_child_at = find_first_child(0, subject.root_children);
172 }
173
174 workspace.history.clear();
175 while (1) {
176 Index_ current_subject;
177 bool skip_search;
178 if (workspace.history.empty()) {
179 if (root_child_at == subject.root_children || is_finished(subject.starts[root_child_at])) {
180 break;
181 }
182 current_subject = root_child_at;
183 skip_search = root_skip_search;
184 ++root_child_at;
185 } else {
186 auto& current_state = workspace.history.back();
187 if (current_state.child_at == current_state.child_end || is_finished(subject.starts[current_state.child_at])) {
188 workspace.history.pop_back();
189 continue;
190 }
191 current_subject = current_state.child_at;
192 skip_search = current_state.skip_search;
193 ++(current_state.child_at); // do this before the emplace_back(), otherwise the history might get reallocated and the reference would be dangling.
194 }
195
196 const auto& current_node = subject.nodes[current_subject];
197 if constexpr(mode_ == OverlapsAnyMode::MIN_OVERLAP) {
198 if (std::min(query_end, subject.ends[current_subject]) - std::max(query_start, subject.starts[current_subject]) < params.min_overlap) {
199 // No point continuing with the children, as all children will by definition have smaller overlaps and cannot satisfy 'min_overlap'.
200 continue;
201 }
202 }
203
204 matches.push_back(current_node.id);
205 if (params.quit_on_first) {
206 return;
207 }
208 if (current_node.duplicates_start != current_node.duplicates_end) {
209 matches.insert(matches.end(), subject.duplicates.begin() + current_node.duplicates_start, subject.duplicates.begin() + current_node.duplicates_end);
210 }
211
212 if (current_node.children_start != current_node.children_end) {
213 if (skip_search) {
214 workspace.history.emplace_back(current_node.children_start, current_node.children_end, true);
215 } else {
216 Index_ start_pos = find_first_child(current_node.children_start, current_node.children_end);
217 if (start_pos != current_node.children_end) {
218 workspace.history.emplace_back(start_pos, current_node.children_end, is_query_preceding(subject.starts[start_pos]));
219 }
220 }
221 }
222 }
